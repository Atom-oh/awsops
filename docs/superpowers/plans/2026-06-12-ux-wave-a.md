```ts
export const dynamic = 'force-dynamic';
// HttpOnly 쿠키는 서버만 지울 수 있다. Cognito 세션도 hosted-UI /logout으로 끊는다
// (logout_uri는 auth.tf:30 logout_urls에 기등록된 https://{APP_DOMAIN}/).
export async function POST() {
  const domain = process.env.COGNITO_DOMAIN ?? '';
  const clientId = process.env.COGNITO_CLIENT_ID ?? '';
  const appDomain = process.env.APP_DOMAIN ?? '';
  const logoutUrl = domain && clientId && appDomain
    ? `https://${domain}/logout?client_id=${encodeURIComponent(clientId)}&logout_uri=${encodeURIComponent(`https://${appDomain}/`)}`
    : '/';
  return new Response(JSON.stringify({ logoutUrl }), {
    status: 200,
    headers: {
      'content-type': 'application/json',
      'set-cookie': 'awsops_token=; Path=/; Secure; HttpOnly; SameSite=Lax; Max-Age=0',
    },
  });
}
```

정찰 사실: `/api/cost`의 monthly는 서비스별 분해를 합산-폐기. **동시 세션의 `getServiceCostDetail` 소스는 유실**(.next 산출물만) — 동일 시그니처로 신규 구현(향후 충돌 최소화): `{service, currency, trend|null, byUsageType|null}` (null=해당 CE 호출 실패, []=정말 빈 것).

- [ ] **Step 1 (TDD): `rollupUsageTypes` 순수 함수** — `aws.test.ts`: 내림차순 정렬, top-n 유지+잔여 '기타' 합산, n 이하면 그대로, 잔여 0이면 '기타' 생략. 구현(`aws.ts`):

```ts
export interface ServiceUsageType { usageType: string; amount: number }
export interface ServiceCostDetail {
  service: string; currency: string;
  trend: CostTrendPoint[] | null;          // null = trend CE 호출 실패 (vs [] = 진짜 빈 것)
  byUsageType: ServiceUsageType[] | null;
}
export function rollupUsageTypes(rows: ServiceUsageType[], n: number): ServiceUsageType[] {
  const sorted = [...rows].sort((a, b) => b.amount - a.amount);
  if (sorted.length <= n) return sorted;
  const head = sorted.slice(0, n);
  const rest = sorted.slice(n).reduce((s, x) => s + x.amount, 0);
  return rest > 0 ? [...head, { usageType: '기타', amount: rest }] : head;
}
```
